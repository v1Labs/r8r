Basic Radar Chart

Simple single dataset visualization

const chart = [ { label: 'Speed', value: 0, maxValue: 100 }, { label: 'Reliability', value: 0, maxValue: 100 }, { label: 'Comfort', value: 0, maxValue: 100 }, { label: 'Safety', value: 0, maxValue: 100 }, { label: 'Efficiency', value: 0, maxValue: 100 } ]; const data = [ { label: 'Car Performance', values: { 'Speed': 85, 'Reliability': 92, 'Comfort': 78, 'Safety': 95, 'Efficiency': 88 } } ]; <R8R data={data} chart={chart} />
